def binaryfuck_to_brainfuck(binary_code):
    bf_mapping = {
        '000': '+',
        '001': '-',
        '010': '>',
        '011': '<',
        '100': '.',
        '101': ',',
        '110': '[',
        '111': ']'
    }
    commands = [binary_code[i:i + 3] for i in range(0, len(binary_code), 3)]
    brainfuck_code = ''.join(bf_mapping[cmd] for cmd in commands if cmd in bf_mapping)
    return brainfuck_code
